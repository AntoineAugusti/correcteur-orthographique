		\tadOperation{dictionnaire}{\tadUnParam{}}{\tadUnParam{\dictionnaire}}
		\tadOperation{estVide}{\tadUnParam{\dictionnaire}}{\tadUnParam{\booleen}}
		\tadOperation{insererMot}{\tadDeuxParams{\dictionnaire}{\mot}}{\tadUnParam{\dictionnaire}}
		\tadOperation{insererFichier}{\tadDeuxParams{\dictionnaire}{\chaine}}{\tadUnParam{\dictionnaire}}

		\tadOperation{sauvegarder}{\tadDeuxParams{\dictionnaire}{\chaine}}{\tadUnParam{\fichierTexte}}

	\end{tadAxiomes}
