	\begin{tadAxiomes}
		\tadAxiome{longueur(insererCaractere(mot, c, i)) = longueur(mot) + 1}
		\tadAxiome{longueur(modifierCaractere(mot, c, i)) = longueur(mot)}
		\tadAxiome{longueur(supprimerCaractere(mot, c, i)) = longueur(mot) - 1}
		\tadAxiome{sontEgaux(supprimerCaractere(insererCaractere(mot, c, i), i), mot)}
		\tadAxiome{sontEgaux(insererCaractere(supprimerCaractere(mot, i), c, i), mot)}
		\tadAxiome{iemeCaractere(insererCaractere(mot, c, i), i) = c}
		\tadAxiome{longueur(concatener(mot1, mot2)) = longueur(mot1) + longueur(mot2)}
		\tadAxiome{iemeCaractere(concatener(mot1, mot2), longueur(mot1) + 1) = iemeCaractere(mot2, 1)}
		\tadAxiome{separerMots(concatener(mot1, mot2), longueur(mot1) + 1) = mot1 , mot2}
	\end{tadAxiomes}
	
	\begin{tadPreconditions}{Mot}
		\tadPrecondition{insererCaractere}{1 \infegalea i \infegalea longueur(mot) + 1 et estLettreOuTiret(c)}

		\tadPrecondition{supprimerCaractere}{1 \infegalea i \infegalea longueur(mot)}
		\tadPrecondition{separerMots}{1 \infegalea i \infegalea longueur(mot)}
		\tadPrecondition{iemeCaractere}{ 1 \infegalea i \infegalea longueur(mot) et non(longueur(mot) = 0)}
		\tadPrecondition{sousMot}{1 \infegalea debut + longueurSousMot \infegalea longueur(mot) et non(longueur(mot) = 0)}
	\end{tadPreconditions}
